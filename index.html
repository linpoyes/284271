// ============================================================ // 易仕門市 — 哈根達斯預購系統 Google Apps Script // 首次執行會自動建立「庫存」「訂單」工作表及標題列 // ============================================================ var FLAVOR_IDS = [ "strawberry", "vanilla_caramel", "matcha", "cookies_cream", "chocolate_almond", "macadamia" ]; var ORDER_HEADERS = [ "單號", "姓名", "電話", "推薦人", "備註", "取貨日期", "口味明細", "數量", "金額", "下單時間", "狀態" ]; // ===================== 自動初始化 ===================== function ensureSheets() { var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- 庫存工作表 --- var stockSheet = ss.getSheetByName("庫存"); if (!stockSheet) { stockSheet = ss.insertSheet("庫存"); // 寫入口味 ID 標題列 stockSheet.getRange(1, 1, 1, FLAVOR_IDS.length).setValues([FLAVOR_IDS]); // 初始庫存設為 0 var zeros = FLAVOR_IDS.map(function() { return 0; }); stockSheet.getRange(2, 1, 1, FLAVOR_IDS.length).setValues([zeros]); // 格式美化 stockSheet.getRange(1, 1, 1, FLAVOR_IDS.length) .setFontWeight("bold") .setBackground("#E1F5EE"); stockSheet.setFrozenRows(1); for (var i = 1; i <= FLAVOR_IDS.length; i++) { stockSheet.autoResizeColumn(i); } } // --- 訂單工作表 --- var orderSheet = ss.getSheetByName("訂單"); if (!orderSheet) { orderSheet = ss.insertSheet("訂單"); // 寫入標題列 orderSheet.getRange(1, 1, 1, ORDER_HEADERS.length).setValues([ORDER_HEADERS]); // 格式美化 orderSheet.getRange(1, 1, 1, ORDER_HEADERS.length) .setFontWeight("bold") .setBackground("#EEEDFE"); orderSheet.setFrozenRows(1); // 口味明細欄拉寬 orderSheet.setColumnWidth(7, 250); for (var j = 1; j <= ORDER_HEADERS.length; j++) { if (j !== 7) orderSheet.autoResizeColumn(j); } } // --- 刪除預設的「工作表1」--- var sheet1 = ss.getSheetByName("工作表1"); if (sheet1 && ss.getSheets().length > 1) { ss.deleteSheet(sheet1); } return { stockSheet: stockSheet, orderSheet: orderSheet }; } // ===================== 進入點 ===================== function doGet(e) { var rawData = e && e.parameter && e.parameter.data ? e.parameter.data : null; try { // 每次請求都確保工作表存在 ensureSheets(); // 無參數 → 讀取全部資料 if (!rawData) { return jsonReply(getAllData()); } var req = JSON.parse(rawData); var action = req.action; if (action === "saveOrder") { return jsonReply(handleSaveOrder(req)); } if (action === "updateStock") { return jsonReply(handleUpdateStock(req)); } if (action === "updateOrders") { return jsonReply(handleUpdateOrders(req)); } if (action === "deleteOrder") { return jsonReply(handleDeleteOrder(req)); } if (action === "updateOrderStatus") { return jsonReply(handleUpdateOrderStatus(req)); } if (action === "updateOrderNote") { return jsonReply(handleUpdateOrderNote(req)); } return jsonReply({ ok: false, error: "unknown action" }); } catch (err) { return jsonReply({ ok: false, error: String(err) }); } } // ===================== 回傳 JSON ===================== function jsonReply(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ===================== 讀取全部資料 ===================== function getAllData() { var lock = LockService.getScriptLock(); lock.waitLock(10000); try { return { ok: true, stock: readStock(), orders: readOrders() }; } finally { lock.releaseLock(); } } // ===================== 儲存訂單（原子扣庫存） ===================== function handleSaveOrder(req) { var order = req.order; var deductions = req.stockDeductions; var lock = LockService.getScriptLock(); lock.waitLock(15000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- 扣庫存 --- var stockSheet = ss.getSheetByName("庫存"); var row = stockSheet.getRange(2, 1, 1, FLAVOR_IDS.length).getValues()[0]; for (var i = 0; i < FLAVOR_IDS.length; i++) { var fid = FLAVOR_IDS[i]; if (deductions[fid]) { row[i] = Math.max(0, (Number(row[i]) || 0) - Number(deductions[fid])); } } stockSheet.getRange(2, 1, 1, FLAVOR_IDS.length).setValues([row]); // --- 寫入訂單 --- var orderSheet = ss.getSheetByName("訂單"); var lastRow = orderSheet.getLastRow(); // 取最大單號 + 1（避免刪除訂單後單號重複） var maxNo = 0; if (lastRow > 1) { var noCol = orderSheet.getRange(2, 1, lastRow - 1, 1).getValues(); for (var n = 0; n < noCol.length; n++) { var num = parseInt(String(noCol[n][0]).replace(/^'+/, ""), 10); if (!isNaN(num) && num > maxNo) maxNo = num; } } var orderNo = String(maxNo + 1).padStart(3, "0"); // 電話前加 ' 讓試算表當純文字存（避免吃掉前導零） var phoneText = order.phone ? "'" + String(order.phone) : ""; // 取貨日期也當純文字存（避免被轉為 Date 物件） var pickupText = order.pickupDate ? "'" + String(order.pickupDate) : ""; orderSheet.appendRow([ "'" + orderNo, order.name || "", phoneText, order.referrer || "", order.note || "", pickupText, JSON.stringify(order.items || {}), order.total || 0, order.amount || 0, order.time || new Date().toLocaleString("zh-TW"), order.status || "pending" ]); var newStock = {}; for (var j = 0; j < FLAVOR_IDS.length; j++) { newStock[FLAVOR_IDS[j]] = Number(row[j]) || 0; } return { ok: true, orderNo: orderNo, stock: newStock }; } finally { lock.releaseLock(); } } // ===================== 覆寫庫存 ===================== function handleUpdateStock(req) { var newStock = req.stock; var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("庫存"); var row = []; for (var i = 0; i < FLAVOR_IDS.length; i++) { row.push(Number(newStock[FLAVOR_IDS[i]]) || 0); } sheet.getRange(2, 1, 1, FLAVOR_IDS.length).setValues([row]); return { ok: true }; } finally { lock.releaseLock(); } } // ===================== 覆寫全部訂單 ===================== function handleUpdateOrders(req) { var orders = req.orders || []; var lock = LockService.getScriptLock(); lock.waitLock(15000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("訂單"); var lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, ORDER_HEADERS.length).clearContent(); } if (orders.length > 0) { var rows = orders.map(function(o, idx) { var ph = o.phone ? "'" + String(o.phone) : ""; var pd = o.pickupDate ? "'" + String(o.pickupDate) : ""; return [ o.orderNo || String(idx + 1).padStart(3, "0"), o.name || "", ph, o.referrer || "", o.note || "", pd, JSON.stringify(o.items || {}), o.total || 0, o.amount || 0, o.time || "", o.status || "pending" ]; }); sheet.getRange(2, 1, rows.length, ORDER_HEADERS.length).setValues(rows); } return { ok: true }; } finally { lock.releaseLock(); } } // ===================== 刪除單筆訂單（by orderNo）並回補庫存 ===================== function handleDeleteOrder(req) { var orderNo = String(req.orderNo); var lock = LockService.getScriptLock(); lock.waitLock(15000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("訂單"); var stockSheet = ss.getSheetByName("庫存"); var lastRow = orderSheet.getLastRow(); if (lastRow <= 1) return { ok: false, error: "no orders" }; var data = orderSheet.getRange(2, 1, lastRow - 1, ORDER_HEADERS.length).getValues(); var targetRow = -1; var items = {}; for (var i = 0; i < data.length; i++) { var sheetNo = String(data[i][0]).replace(/^0+/, "") || "0"; var reqNo = orderNo.replace(/^0+/, "") || "0"; if (sheetNo === reqNo) { targetRow = i + 2; try { items = JSON.parse(data[i][6]); } catch(e) {} break; } } if (targetRow === -1) return { ok: false, error: "order not found" }; // 刪除該列 orderSheet.deleteRow(targetRow); // 回補庫存 var stockRow = stockSheet.getRange(2, 1, 1, FLAVOR_IDS.length).getValues()[0]; for (var j = 0; j < FLAVOR_IDS.length; j++) { var fid = FLAVOR_IDS[j]; if (items[fid]) { stockRow[j] = (Number(stockRow[j]) || 0) + Number(items[fid]); } } stockSheet.getRange(2, 1, 1, FLAVOR_IDS.length).setValues([stockRow]); return { ok: true, stock: readStock(), orders: readOrders() }; } finally { lock.releaseLock(); } } // ===================== 更新訂單狀態（by orderNo） ===================== function handleUpdateOrderStatus(req) { var orderNo = String(req.orderNo); var newStatus = req.status || "pending"; var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("訂單"); var lastRow = sheet.getLastRow(); if (lastRow <= 1) return { ok: false, error: "no orders" }; var data = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); var reqNo = orderNo.replace(/^0+/, "") || "0"; for (var i = 0; i < data.length; i++) { var sheetNo = String(data[i][0]).replace(/^0+/, "") || "0"; if (sheetNo === reqNo) { sheet.getRange(i + 2, 11).setValue(newStatus); return { ok: true }; } } return { ok: false, error: "order not found" }; } finally { lock.releaseLock(); } } // ===================== 更新訂單備註（by orderNo） ===================== function handleUpdateOrderNote(req) { var orderNo = String(req.orderNo); var note = req.adminNote || ""; var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("訂單"); var lastRow = sheet.getLastRow(); if (lastRow <= 1) return { ok: false, error: "no orders" }; var data = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); var reqNo = orderNo.replace(/^0+/, "") || "0"; for (var i = 0; i < data.length; i++) { var sheetNo = String(data[i][0]).replace(/^0+/, "") || "0"; if (sheetNo === reqNo) { sheet.getRange(i + 2, 5).setValue(note); return { ok: true }; } } return { ok: false, error: "order not found" }; } finally { lock.releaseLock(); } } // ===================== 讀取庫存 ===================== function readStock() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("庫存"); var row = sheet.getRange(2, 1, 1, FLAVOR_IDS.length).getValues()[0]; var stock = {}; for (var i = 0; i < FLAVOR_IDS.length; i++) { stock[FLAVOR_IDS[i]] = Number(row[i]) || 0; } return stock; } // ===================== 讀取訂單 ===================== function readOrders() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("訂單"); var lastRow = sheet.getLastRow(); if (lastRow <= 1) return []; var data = sheet.getRange(2, 1, lastRow - 1, ORDER_HEADERS.length).getValues(); var orders = []; for (var i = 0; i < data.length; i++) { var r = data[i]; if (!r[0] && !r[1]) continue; var items = {}; try { items = JSON.parse(r[6]); } catch(e) {} // 電話號碼：補前導零（試算表可能存成數字吃掉0） var rawPhone = String(r[2]).replace(/'/g, ""); if (/^\d{9}$/.test(rawPhone)) rawPhone = "0" + rawPhone; // 取貨日期：如果被試算表轉成 Date 物件，格式化回 M/D var rawPickup = r[5]; if (rawPickup instanceof Date && !isNaN(rawPickup.getTime())) { rawPickup = (rawPickup.getMonth() + 1) + "/" + rawPickup.getDate(); } else { rawPickup = String(rawPickup).replace(/'/g, ""); } orders.push({ id: "gas_" + String(r[0]) + "_" + i, orderNo: String(r[0]), name: String(r[1]), phone: rawPhone, referrer: String(r[3]), note: String(r[4]), pickupDate: rawPickup, items: items, total: Number(r[7]) || 0, amount: Number(r[8]) || 0, time: String(r[9]), status: String(r[10]) || "pending" }); } return orders; }
